    let cart = {};
    function updateCart(name, price, quantity) {
      cart[name] = (cart[name] || 0) + quantity;
      if (cart[name] <= 0) delete cart[name];
      renderCart();
    }

    function renderCart() {
      const cartEl = document.getElementById('cart-values');
      cartEl.innerHTML = '';
      let total = 0;
      for (const item in cart) {
        const quantity = cart[item];
        const price = items[item];
        total += quantity * price;
        cartEl.innerHTML += `<li>${item} x ${quantity} = ₹${quantity * price}</li>`;
      }
      if (total === 0) cartEl.innerHTML = '<li>Cart is empty</li>';
      else cartEl.innerHTML += `<li><strong>Total: ₹${total}</strong></li>`;
    }

    const items = {
